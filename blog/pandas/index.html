常用功能

  • Transport: 只要invoke df.T即可得到轉置後的結果
import pandas as pd
import numpy as np

df = pd.DataFrame(np.random.randn(10, 4), columns=["A", "B", "C", "D"])
print(df)
# 取得index 0的row
print(df[0:1])
# 取得一整個column A的值
print(df["A"])
# 對df做轉置
print(df.T)
# 取得column2的值
print(df.T[0])
# 取多個column值, 可以透過iloc取值(如果columns是數字)
print(df.T.iloc[:, 0:1])
# 注意！ 此時由於index為Ａ~Ｄ, 所以沒辦法用String取值, 所以得用數字取值
print(df.T[0:3])
# 原本的dataframe
print(df.T.T)
